162  private void FireMessageReceivedEvent(MessageEnvelope messageEnvelope)
163  {
164  var task = new Task(() =>
165  {
166  var eventArgs = new MessageReceivedEventArgs(messageEnvelope);
167 
168  MessageReceived?.Invoke(this, eventArgs);
169 
170  if (!autoAcknowledgeMessages)
171  {
172  if (eventArgs.Response == MessageResponse.Ack)
173  {
174  Channel.BasicAck(messageEnvelope.DeliveryTag, false);
175  }
176  else if (eventArgs.Response == MessageResponse.Nack)
177  {
178  Channel.BasicAck(messageEnvelope.DeliveryTag, false);
179  }
180  else if (eventArgs.Response == MessageResponse.RejectPermanent)
181  {
182  Channel.BasicReject(messageEnvelope.DeliveryTag, false);
183  }
184  else
185  {
186  Channel.BasicReject(messageEnvelope.DeliveryTag, true);
187  }
188  }
189 
190  limitedConsumeSempahore?.Give();
191  });
192 
193  task.Start();
194  }
