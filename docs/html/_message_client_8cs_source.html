175  private void DedicatedQueue_MessageReceived(object sender, MessageReceivedEventArgs e)
176  {
177  //var signatureIsValid = clientServerHandshakeComplete && SignatureIsValid(e.RawData, PublicKeystore[e.Message.ClientName].Dsa);
178 
179  Log.Verbose(e.MessageEnevelope.GetType().ToString());
180 
181  if (e.MessageEnevelope.BasicProperties.CorrelationId != null &&
182  rpcCallWaitInfo.ContainsKey(e.MessageEnevelope.BasicProperties.CorrelationId))
183  {
184  var waitInfo = rpcCallWaitInfo[e.MessageEnevelope.BasicProperties.CorrelationId];
185  waitInfo.ResponseMessageEnvelope = e.MessageEnevelope;
186  //waitInfo.RawResponse = e.RawData;
187  waitInfo.WaitHandle.Set();
188  }
189  else if (e.MessageEnevelope.SignatureVerificationStatus == SignatureVerificationStatus.SignatureValid)
190  {
191  if (e.MessageEnevelope.GetType() == typeof(ClientReannounceRequestMessage))
192  {
193  Log.Information("Server requested reannouncement of clients.");
194  var message = new ClientAnnounceMessage(PublicKeystore.ParentClientInfo.ECKey);
195 
196  //todo: Encrypt?
197  WriteToServerNoWait(message, false);
198  }
199  else if (e.MessageEnevelope.GetType() == typeof(HeartbeatMessage))
200  {
201  if (!serverHeartbeatTimer.Enabled)
202  {
203  Log.Information("Server heartbeat re-established.");
204  }
205 
206  serverHeartbeatTimer.Restart();
207  }
208  else if (e.MessageEnevelope.GetType() == typeof(PublicKeyUpdate))
209  {
210  var tempMessage = (PublicKeyUpdate)e.MessageEnevelope.Message;
211  PublicKeystore.Add(tempMessage.ClientInfo);
212  Log.Verbose($"Received public key '{tempMessage.ClientInfo.ECKey.GetPublicKeyHash()}' for client '{tempMessage.ClientInfo.Name}'.");
213  }
214  else if (e.MessageEnevelope.GetType() == typeof(SystemSharedKeyUpdate))
215  {
216  throw new NotImplementedException();
217  }
218  else
219  {
220  FireMessageReceivedEvent(e);
221  }
222  }
223  else if (clientServerHandshakeComplete)
224  {
225  throw new Exception();
226  }
227  }
