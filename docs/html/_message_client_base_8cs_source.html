356  protected async Task<RpcCallWaitInfo> WriteToClient(string clientName, MessageBase message, int waitTime, EncryptionOption encryptionOptions)
357  {
358  if (message == null)
359  {
360  throw new ArgumentNullException(nameof(message));
361  }
362 
363  var formattedMessage = ConfigureUnicastMessageForSend(message, clientName, encryptionOptions);
364 
365  Channel.BasicPublish(
366  exchange: "",
367  routingKey: NameHelper.GetDedicatedQueueName(SystemName, clientName),
368  basicProperties: formattedMessage.BasicProperties,
369  formattedMessage.SignedMessage);
370 
371  RpcCallWaitInfo response = new RpcCallWaitInfo();
372  response.WaitHandle = new EventWaitHandle(false, EventResetMode.ManualReset);
373 
374  rpcCallWaitInfo.Add(formattedMessage.BasicProperties.CorrelationId, response);
375 
376  Task task = new Task(() =>
377  {
378  if (!response.WaitHandle.WaitOne(waitTime))
379  {
380  response.CallResult = RpcCallResult.Timeout;
381  }
382  else
383  {
384  if (response.ResponseMessageEnvelope.SignatureVerificationStatus == SignatureVerificationStatus.SignatureValid)
385  {
386  response.CallResult = RpcCallResult.Success;
387  }
388  else
389  {
390  response.CallResult = RpcCallResult.BadSignature;
391  }
392  }
393 
394  rpcCallWaitInfo.Remove(formattedMessage.BasicProperties.CorrelationId);
395  response.WaitHandle.Dispose();
396  });
397 
398  task.Start();
399 
400  await task.ConfigureAwait(false);
401 
402  return response;
403  }
