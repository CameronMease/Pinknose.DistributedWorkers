116  public byte[] Serialize()
117  {
118  byte[] senderBytes = Encoding.UTF8.GetBytes(SenderName);
119  byte[] iv = Array.Empty<byte>();
120  int sharedKeyId = 0;
121 
122  var messageBytes = Message.Serialize();
123 
124  // Add encryption, if required;
125  if (this._encryptionOption == EncryptionOption.EncryptWithPrivateKey)
126  {
127  (messageBytes, iv) = this._messageClient.EncryptDataWithClientKey(messageBytes, this.RecipientName);
128  }
129  else if (this._encryptionOption == EncryptionOption.EncryptWithSystemSharedKey)
130  {
131  (messageBytes, iv, sharedKeyId) = this._messageClient.EncryptDataWithSystemSharedKey(messageBytes);
132  }
133 
134  int totalBufferSize = 8 + senderBytes.Length + messageBytes.Length + iv.Length + SignatureLengthConst;
135 
136  byte[] bytes = new byte[totalBufferSize];
137 
138  // Load bits
139  var bits = new BitVector32(0);
140  bits[isClientAnnounceMask] = this.Message.GetType() == typeof(ClientAnnounceMessage);
141  bits[isClientAnnounceResponseMask] = this.Message.GetType() == typeof(ClientAnnounceResponseMessage);
142  BitConverter.GetBytes(bits.Data).CopyTo(bytes, 0);
143 
144  // Load numbers
145  var numbers = new BitVector32(0);
146  numbers[signatureLengthSection] = SignatureLengthConst;
147  numbers[encryptionOptionSection] = (int)_encryptionOption;
148  numbers[senderNameLengthSection] = SenderName.Length;
149  numbers[ivLengthSection] = iv.Length;
150  numbers[sharedKeyIdSection] = sharedKeyId;
151  BitConverter.GetBytes(numbers.Data).CopyTo(bytes, 4);
152 
153  // Copy fields
154  senderBytes.CopyTo(bytes, 8);
155  iv.CopyTo(bytes, 8 + senderBytes.Length);
156  messageBytes.CopyTo(bytes, 8 + senderBytes.Length + iv.Length);
157 
158  byte[] signature = _messageClient.SignData(bytes, 0, totalBufferSize - SignatureLengthConst);
159 
160  if (signature.Length != SignatureLengthConst)
161  {
162  // Made bad assumption of signature length!
163  throw new NotImplementedException();
164  }
165 
166  signature.CopyTo(bytes, totalBufferSize - signature.Length);
167 
168  return bytes;
169  }

220  private static MessageEnvelope Deserialize(byte[] bytes, MessageClientBase messageClient)
221  {
222  try
223  {
224  var envelope = new MessageEnvelope();
225 
226  var bits = new BitVector32(BitConverter.ToInt32(bytes, 0));
227  var isClientAnnounce = bits[isClientAnnounceMask];
228  var isClientAnnounceResponse = bits[isClientAnnounceResponseMask];
229 
230  var numbers = new BitVector32(BitConverter.ToInt32(bytes, 4));
231  var signatureLength = numbers[signatureLengthSection];
232  envelope._encryptionOption = (EncryptionOption)numbers[encryptionOptionSection];
233  var senderNameLength = numbers[senderNameLengthSection];
234  var ivLength = numbers[ivLengthSection];
235  var sharedKeyId = numbers[sharedKeyIdSection];
236 
237  int index = 8;
238 
239  envelope.SenderName = Encoding.UTF8.GetString(bytes, index, senderNameLength);
240  index += senderNameLength;
241 
242  var iv = bytes[(index)..(index + ivLength)];
243  index += ivLength;
244 
245  var signature = bytes[^signatureLength..];
246 
247  if (signatureLength != SignatureLengthConst)
248  {
249  throw new NotImplementedException("Signautre length is invalid.");
250  }
251 
252  envelope.SignatureVerificationStatus = messageClient.ValidateSignature(bytes[..^signatureLength], signature, envelope.SenderName);
253 
254  if (envelope.SignatureVerificationStatus != SignatureVerificationStatus.SignatureValid)
255  {
256  throw new NotImplementedException(envelope.SignatureVerificationStatus.ToString());
257  }
258 
259  // Get the bytes representing the message
260  byte[] messageBytes = bytes[index..^signatureLength];
261 
262  // Decrypt the bytes if necessary
263  if (envelope._encryptionOption == EncryptionOption.EncryptWithPrivateKey)
264  {
265  messageBytes = messageClient.DecryptDataWithClientKey(messageBytes, envelope.SenderName, iv);
266  }
267  else if (envelope._encryptionOption == EncryptionOption.EncryptWithSystemSharedKey)
268  {
269  messageBytes = messageClient.DecryptDataWithSystemSharedKey(messageBytes, iv);
270  }
271 
272  // Deserialize the message
273  envelope.Message = MessageBase.Deserialize(messageBytes);
274 
275  return envelope;
276  }
277  catch (Exception e)
278  {
279  throw;
280  }
281  }
